/* ---- animated counter (the "live" catchy line) ---- */
const counter = document.getElementById('counter');
const target = +counter.dataset.target;
const io = new IntersectionObserver((entries) => {
  entries.forEach(e => {
    if (!e.isIntersecting) return;
    const reduce = matchMedia('(prefers-reduced-motion: reduce)').matches;
    if (reduce) { counter.textContent = target; io.disconnect(); return; }
    let n = 0; const step = Math.max(1, Math.round(target / 45));
    const t = setInterval(() => {
      n += step;
      if (n >= target) { n = target; clearInterval(t); }
      counter.textContent = n;
    }, 28);
    io.disconnect();
  });
}, { threshold: 0.5 });
io.observe(counter);

/* ---- tabbed video player: landscape/MP4 plays inline, Facebook reels open a popup ---- */

const isFacebook = u => /facebook\.com/.test(u);
const isFile = u => /\.(mp4|webm|ogg|ogv|mov|m4v)(\?|#|$)/i.test(u);

/* landscape video / YouTube -> plays inside the on-page 16:9 frame */
function playInline(v) {
  const url = v.url.trim();
  playBtn.style.display = 'none';
  stage.querySelector('.poster').style.display = 'none';
  if (isFile(url)) {
    const vid = document.createElement('video');
    vid.src = url; vid.controls = true; vid.autoplay = true; vid.playsInline = true;
    stage.appendChild(vid);
  } else {
    let src = url;
    if (/youtube\.com|youtu\.be/.test(src)) src += (src.includes('?') ? '&' : '?') + 'autoplay=1';
    const f = document.createElement('iframe');
    f.src = src; f.allow = 'autoplay; fullscreen; encrypted-media; picture-in-picture'; f.allowFullscreen = true;
    stage.appendChild(f);
  }
}

/* vertical content (portrait MP4 or Facebook reel) -> portrait popup */
function openPopup(v) {
  const url = v.url.trim();
  lbTitle.textContent = v.title;
  lbMedia.classList.remove('wide');
  lbMedia.innerHTML = '';
  let media;
  if (isFacebook(url)) {
    media = document.createElement('iframe');
    media.allow = 'autoplay; fullscreen; encrypted-media; picture-in-picture; clipboard-write; web-share';
    media.allowFullscreen = true; media.setAttribute('scrolling', 'no');
    media.src = 'https://www.facebook.com/plugins/video.php?href=' +
                encodeURIComponent(url) + '&show_text=false&width=360&height=640&autoplay=true';
  } else {
    media = document.createElement('video');

  }

  lightbox.classList.add('open');
  lightbox.setAttribute('aria-hidden', 'false');
  document.body.style.overflow = 'hidden';
}

function setInvalid(id, bad) {
  document.getElementById(id).classList.toggle('invalid', bad);

function validate() {
  const name = document.getElementById('name').value.trim();
  const company = document.getElementById('company').value.trim();
  const size = document.getElementById('size').value;
  const phone = document.getElementById('phone').value.trim();
  const email = document.getElementById('email').value.trim();
  const picked = chips.filter(c => c.classList.contains('on'));

  const phoneOk = /^[+]?[\d\s().-]{7,}$/.test(phone);
  const emailOk = email === '' || /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);

  setInvalid('f-name', !name);
  setInvalid('f-company', !company);
  setInvalid('f-size', !size);
  setInvalid('f-phone', !phoneOk);
  setInvalid('f-email', !emailOk);
  setInvalid('f-services', picked.length === 0);

  return name && company && size && phoneOk && emailOk && picked.length > 0;
}

    document.getElementById(map[id]).classList.remove('invalid');
  });
});

form.addEventListener('submit', async (e) => {
  e.preventDefault();
  if (!validate()) {
    form.querySelector('.invalid')?.scrollIntoView({ behavior:'smooth', block:'center' });
    return;
  }

  const payload = {
    name: document.getElementById('name').value.trim(),
    company: document.getElementById('company').value.trim(),
    companySize: document.getElementById('size').value,
    phone: document.getElementById('phone').value.trim(),
    email: document.getElementById('email').value.trim(),
    services: chips.filter(c => c.classList.contains('on')).map(c => c.dataset.val),
    details: document.getElementById('details').value.trim(),
    submittedAt: new Date().toISOString()
  };

  submitBtn.disabled = true;
  submitLabel.textContent = 'Sending…';

  try {
    if (CONFIG.formEndpoint) {
      const res = await fetch(CONFIG.formEndpoint, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        body: JSON.stringify(payload)
      });
      if (!res.ok) throw new Error('Request failed');
    } else {
      // preview mode — no backend wired yet
      await new Promise(r => setTimeout(r, 700));
      console.log('Lead captured (preview mode):', payload);
    }
    showSuccess(payload);
  } catch (err) {
    submitBtn.disabled = false;
    submitLabel.textContent = 'Try again';
    alert('Something went wrong sending your details. Please try again, or call us directly.');
  }
});

}
